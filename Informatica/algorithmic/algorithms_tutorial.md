### üì¶ –ú–∞—Å—Å–∏–≤—ã (–°–ø–∏—Å–∫–∏ –≤ Python)

**–ú–∞—Å—Å–∏–≤** ‚Äî —ç—Ç–æ –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç—å —ç–ª–µ–º–µ–Ω—Ç–æ–≤ –æ–¥–Ω–æ–≥–æ —Ç–∏–ø–∞, —Ä–∞—Å–ø–æ–ª–æ–∂–µ–Ω–Ω—ã—Ö –≤ –ø–∞–º—è—Ç–∏ –¥—Ä—É–≥ –∑–∞ –¥—Ä—É–≥–æ–º.

```python
class DynamicArray:
    """–£—á–µ–±–Ω–∞—è —Ä–µ–∞–ª–∏–∑–∞—Ü–∏—è –¥–∏–Ω–∞–º–∏—á–µ—Å–∫–æ–≥–æ –º–∞—Å—Å–∏–≤–∞"""
    
    def __init__(self):
        self.capacity = 2  # –ù–∞—á–∞–ª—å–Ω–∞—è –≤–º–µ—Å—Ç–∏–º–æ—Å—Ç—å
        self.length = 0    # –¢–µ–∫—É—â–∞—è –¥–ª–∏–Ω–∞
        self.array = [None] * self.capacity
    
    def __getitem__(self, index):
        if 0 <= index < self.length:
            return self.array[index]
        raise IndexError("–ò–Ω–¥–µ–∫—Å –≤–Ω–µ –≥—Ä–∞–Ω–∏—Ü –º–∞—Å—Å–∏–≤–∞")
    
    def append(self, value):
        """–î–æ–±–∞–≤–ª–µ–Ω–∏–µ —ç–ª–µ–º–µ–Ω—Ç–∞ –≤ –∫–æ–Ω–µ—Ü"""
        if self.length == self.capacity:
            self._resize()  # –£–≤–µ–ª–∏—á–∏–≤–∞–µ–º –º–∞—Å—Å–∏–≤ –ø—Ä–∏ –Ω–µ–æ–±—Ö–æ–¥–∏–º–æ—Å—Ç–∏
        
        self.array[self.length] = value
        self.length += 1
    
    def _resize(self):
        """–£–≤–µ–ª–∏—á–µ–Ω–∏–µ —Ä–∞–∑–º–µ—Ä–∞ –º–∞—Å—Å–∏–≤–∞ –≤ 2 —Ä–∞–∑–∞"""
        new_capacity = self.capacity * 2
        new_array = [None] * new_capacity
        
        for i in range(self.length):
            new_array[i] = self.array[i]
        
        self.array = new_array
        self.capacity = new_capacity
    
    def __str__(self):
        return str([self.array[i] for i in range(self.length)])

# üß™ –î–µ–º–æ–Ω—Å—Ç—Ä–∞—Ü–∏—è —Ä–∞–±–æ—Ç—ã
```

```python
class Node:
    """–£–∑–µ–ª —Å–≤—è–∑–Ω–æ–≥–æ —Å–ø–∏—Å–∫–∞"""
    def __init__(self, data):
        self.data = data
        self.next = None
    
    def __repr__(self):
        return f"Node({self.data})"

class LinkedList:
    """–û–¥–Ω–æ—Å–≤—è–∑–Ω—ã–π —Å–ø–∏—Å–æ–∫"""
    
    def __init__(self):
        self.head = None
        self.tail = None
        self.length = 0
    
    def append(self, data):
        """–î–æ–±–∞–≤–ª–µ–Ω–∏–µ –≤ –∫–æ–Ω–µ—Ü —Å–ø–∏—Å–∫–∞"""
        new_node = Node(data)
        
        if self.head is None:  # –ï—Å–ª–∏ —Å–ø–∏—Å–æ–∫ –ø—É—Å—Ç
            self.head = new_node
            self.tail = new_node
        else:  # –ï—Å–ª–∏ –≤ —Å–ø–∏—Å–∫–µ —É–∂–µ –µ—Å—Ç—å —ç–ª–µ–º–µ–Ω—Ç—ã
            self.tail.next = new_node
            self.tail = new_node
        
        self.length += 1
    
    def prepend(self, data):
        """–î–æ–±–∞–≤–ª–µ–Ω–∏–µ –≤ –Ω–∞—á–∞–ª–æ —Å–ø–∏—Å–∫–∞"""
        new_node = Node(data)
        
        if self.head is None:
            self.head = new_node
            self.tail = new_node
        else:
            new_node.next = self.head
            self.head = new_node
        
        self.length += 1
    
    def insert(self, index, data):
        """–í—Å—Ç–∞–≤–∫–∞ –ø–æ –∏–Ω–¥–µ–∫—Å—É"""
        if index <= 0:
            self.prepend(data)
        elif index >= self.length:
            self.append(data)
        else:
            new_node = Node(data)
            current = self.head
            
            for _ in range(index - 1):
                current = current.next
            
            new_node.next = current.next
            current.next = new_node
            self.length += 1
    
    def visualize(self):
        """–í–∏–∑—É–∞–ª–∏–∑–∞—Ü–∏—è —Å–ø–∏—Å–∫–∞"""
        result = []
        current = self.head
        
        while current:
            result.append(f"({current.data})")
            if current.next:
                result.append("‚Üí")
            current = current.next
        
        return " ".join(result)

# üé® –°–æ–∑–¥–∞–Ω–∏–µ –∏ –≤–∏–∑—É–∞–ª–∏–∑–∞—Ü–∏—è —Å–ø–∏—Å–∫–∞
llist = LinkedList()
for value in [10, 20, 30, 40, 50]:
    llist.append(value)

print("–°–≤—è–∑–Ω—ã–π —Å–ø–∏—Å–æ–∫:")
print(llist.visualize())
print(f"–î–ª–∏–Ω–∞ —Å–ø–∏—Å–∫–∞: {llist.length}")

# –í—Å—Ç–∞–≤–∫–∞ –Ω–æ–≤–æ–≥–æ —ç–ª–µ–º–µ–Ω—Ç–∞

print(llist.visualize())
```

```python
def binary_search_interactive(arr, target):
    """
    –ò–Ω—Ç–µ—Ä–∞–∫—Ç–∏–≤–Ω—ã–π –±–∏–Ω–∞—Ä–Ω—ã–π –ø–æ–∏—Å–∫ —Å –≤—ã–≤–æ–¥–æ–º –∫–∞–∂–¥–æ–≥–æ —à–∞–≥–∞
    """

    right = len(arr) - 1
    step = 1

    while left <= right:
        mid = (left + right) // 2
        print(f"\n–®–∞–≥ {step}:")
        print(f"  –õ–µ–≤–∞—è –≥—Ä–∞–Ω–∏—Ü–∞: –∏–Ω–¥–µ–∫—Å {left} = {arr[left]}")
        print(f"  –ü—Ä–∞–≤–∞—è –≥—Ä–∞–Ω–∏—Ü–∞: –∏–Ω–¥–µ–∫—Å {right} = {arr[right]}")
        print(f"  –°–µ—Ä–µ–¥–∏–Ω–∞: –∏–Ω–¥–µ–∫—Å {mid} = {arr[mid]}")
        
        if arr[mid] == target:
            print(f"  üéâ –≠–ª–µ–º–µ–Ω—Ç –Ω–∞–π–¥–µ–Ω –Ω–∞ –ø–æ–∑–∏—Ü–∏–∏ {mid}!")
            return mid
        elif arr[mid] < target:
            print(f"  {arr[mid]} < {target}, –∏—â–µ–º —Å–ø—Ä–∞–≤–∞")
            left = mid + 1
        else:
            print(f"  {arr[mid]} > {target}, –∏—â–µ–º —Å–ª–µ–≤–∞")
            right = mid - 1
        
        step += 1
    
    print("\n‚ùå –≠–ª–µ–º–µ–Ω—Ç –Ω–µ –Ω–∞–π–¥–µ–Ω")
    return -1
```

```python
    """
```

```python
def quick_sort_explained(arr, depth=0):
    """
    –ë—ã—Å—Ç—Ä–∞—è —Å–æ—Ä—Ç–∏—Ä–æ–≤–∫–∞ —Å –æ–±—ä—è—Å–Ω–µ–Ω–∏–µ–º –∫–∞–∂–¥–æ–≥–æ —à–∞–≥–∞
    """
    indent = "  " * depth
    
    if len(arr) <= 1:
        print(f"{indent}–ë–∞–∑–æ–≤—ã–π —Å–ª—É—á–∞–π: {arr} (–¥–ª–∏–Ω–∞ ‚â§ 1)")
        return arr
    
    pivot = arr[len(arr) // 2]
    print(f"{indent}–û–ø–æ—Ä–Ω—ã–π —ç–ª–µ–º–µ–Ω—Ç: {pivot}")
    print(f"{indent}–ú–∞—Å—Å–∏–≤ –¥–ª—è —Å–æ—Ä—Ç–∏—Ä–æ–≤–∫–∏: {arr}")
    
    left = [x for x in arr if x < pivot]
    middle = [x for x in arr if x == pivot]
    right = [x for x in arr if x > pivot]
    
    print(f"{indent}–†–∞–∑–¥–µ–ª–µ–Ω–∏–µ:")
    print(f"{indent}  –õ–µ–≤–∞—è —á–∞—Å—Ç—å (< {pivot}): {left}")
    print(f"{indent}  –°—Ä–µ–¥–Ω—è—è —á–∞—Å—Ç—å (= {pivot}): {middle}")
    print(f"{indent}  –ü—Ä–∞–≤–∞—è —á–∞—Å—Ç—å (> {pivot}): {right}")
    
    # –†–µ–∫—É—Ä—Å–∏–≤–Ω–∞—è —Å–æ—Ä—Ç–∏—Ä–æ–≤–∫–∞
    sorted_left = quick_sort_explained(left, depth + 1)
    sorted_right = quick_sort_explained(right, depth + 1)
    
    result = sorted_left + middle + sorted_right
    print(f"{indent}–û–±—ä–µ–¥–∏–Ω–µ–Ω–∏–µ: {sorted_left} + {middle} + {sorted_right} = {result}")
    
    return result

# üìö –ü—Ä–∏–º–µ—Ä
```

```python
def dfs_recursive(graph, start, visited=None, path=None):
```
