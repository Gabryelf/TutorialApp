        self.left = None          # –õ–µ–≤—ã–π –ø–æ—Ç–æ–º–æ–∫ (–ø–µ—Å–Ω–∏, –∫–æ—Ç–æ—Ä—ã–µ –∏–¥—É—Ç "—Ä–∞–Ω—å—à–µ")
        self.right = None         # –ü—Ä–∞–≤—ã–π –ø–æ—Ç–æ–º–æ–∫ (–ø–µ—Å–Ω–∏, –∫–æ—Ç–æ—Ä—ã–µ –∏–¥—É—Ç "–ø–æ–∑–∂–µ")
    
    def __str__(self):
        return str(self.song)

class SongTree:
    """
    –ë–∏–Ω–∞—Ä–Ω–æ–µ –¥–µ—Ä–µ–≤–æ –ø–æ–∏—Å–∫–∞ –¥–ª—è –±—ã—Å—Ç—Ä–æ–≥–æ –ø–æ–∏—Å–∫–∞ –ø–µ—Å–µ–Ω –ø–æ –Ω–∞–∑–≤–∞–Ω–∏—é
    """
    
    def __init__(self):
        self.root = None  # –ö–æ—Ä–µ–Ω—å –¥–µ—Ä–µ–≤–∞
    
    def insert(self, song):
        """
        –í—Å—Ç–∞–≤–ª—è–µ—Ç –ø–µ—Å–Ω—é –≤ –¥–µ—Ä–µ–≤–æ (–ø–æ –∞–ª—Ñ–∞–≤–∏—Ç—É –Ω–∞–∑–≤–∞–Ω–∏—è)
        """
        if self.root is None:
            self.root = TreeNode(song)
        else:
            self._insert_recursive(self.root, song)
    
    def _insert_recursive(self, node, song):
        """
        –†–µ–∫—É—Ä—Å–∏–≤–Ω–∞—è –≤—Å—Ç–∞–≤–∫–∞ –ø–µ—Å–Ω–∏ –≤ –¥–µ—Ä–µ–≤–æ
        """
        # –°—Ä–∞–≤–Ω–∏–≤–∞–µ–º –Ω–∞–∑–≤–∞–Ω–∏—è –ø–µ—Å–µ–Ω
        if song.title.lower() < node.song.title.lower():
            # –ï—Å–ª–∏ –Ω–æ–≤–∞—è –ø–µ—Å–Ω—è –¥–æ–ª–∂–Ω–∞ –±—ã—Ç—å —Å–ª–µ–≤–∞
            if node.left is None:
                node.left = TreeNode(song)
            else:
                self._insert_recursive(node.left, song)
        else:
            # –ï—Å–ª–∏ –Ω–æ–≤–∞—è –ø–µ—Å–Ω—è –¥–æ–ª–∂–Ω–∞ –±—ã—Ç—å —Å–ø—Ä–∞–≤–∞
            if node.right is None:
                node.right = TreeNode(song)
            else:
                self._insert_recursive(node.right, song)
    
    def search(self, title):
        """
        –ò—â–µ—Ç –ø–µ—Å–Ω—é –ø–æ –Ω–∞–∑–≤–∞–Ω–∏—é –≤ –¥–µ—Ä–µ–≤–µ
        """
        return self._search_recursive(self.root, title.lower())
    
    def _search_recursive(self, node, title):
        """
        –†–µ–∫—É—Ä—Å–∏–≤–Ω—ã–π –ø–æ–∏—Å–∫ –ø–µ—Å–Ω–∏
        """
        if node is None:
            return None
        
        if title == node.song.title.lower():
            return node.song
        elif title < node.song.title.lower():
            return self._search_recursive(node.left, title)
        else:
            return self._search_recursive(node.right, title)
    
    def display_tree(self, level=0, prefix="–ö–æ—Ä–µ–Ω—å:"):
        """
        –ö—Ä–∞—Å–∏–≤–æ –æ—Ç–æ–±—Ä–∞–∂–∞–µ—Ç –¥–µ—Ä–µ–≤–æ –≤ –∫–æ–Ω—Å–æ–ª–∏
        """
        if self.root is None:
            print("–î–µ—Ä–µ–≤–æ –ø—É—Å—Ç–æ–µ")
            return
        
        self._display_recursive(self.root, level, prefix)
    
    def _display_recursive(self, node, level=0, prefix=""):
        """
        –†–µ–∫—É—Ä—Å–∏–≤–Ω—ã–π –≤—ã–≤–æ–¥ –¥–µ—Ä–µ–≤–∞
        """
        if node is not None:
            indent = "  " * level
            print(f"{indent}{prefix} {node.song.title}")
            
            if node.left or node.right:
                if node.left:
                    self._display_recursive(node.left, level + 1, "‚îú‚îÄ‚îÄ L:")
                if node.right:
                    self._display_recursive(node.right, level + 1, "‚îî‚îÄ‚îÄ R:")
